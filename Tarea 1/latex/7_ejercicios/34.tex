\subsection{EXAMPLE 6.2: FRM EXAM 2009---QUESTION 4-9 (highest EAR)}

\textbf{Enunciado}

\emph{For alternative one-year certificates of deposit
(CDs), CD1 monthly 7.82\%; CD2 quarterly 8.00\%; CD3 semiannually 8.05\%; CD4 continuous 7.95\%. Which CD has the highest EAR?}
\begin{enumerate}[label=\alph*.]
\item CD1
\item CD2
\item CD3
\item \hl{CD4}
\end{enumerate}

\subsubsection*{Solución}


\[
\mathrm{EAR}=\left(1+\frac{r}{m}\right)^{mt}-1,\qquad \mathrm{EAR}_{cont}=e^{rt}-1.
\]







\textbf{CD1:} tasa nominal anual \(7.82\%\) con capitalización mensual.

\[
\mathrm{EAR}_1=
\left(1+\frac{0.0782}{12}\right)^{12}-1
\approx 0.0810646.
\]

\textbf{CD2:} tasa nominal anual \(8.00\%\) con capitalización trimestral.

\[
\mathrm{EAR}_2=
\left(1+\frac{0.08}{4}\right)^4-1
=
(1.02)^4-1
\approx 0.0824322.
\]


\textbf{CD3:} tasa nominal anual \(8.05\%\) con capitalización semestral.


\[
\mathrm{EAR}_3=
\left(1+\frac{0.0805}{2}\right)^2-1
=
(1.04025)^2-1
\approx 0.0821201.
\]


\textbf{CD4:} tasa nominal anual \(7.95\%\) con capitalización continua.

\[
\mathrm{EAR}_4=e^{0.0795}-1\approx 0.0827456.
\]



Reuniendo los resultados:
\[
\mathrm{EAR}_1\approx 8.11\%,\qquad
\mathrm{EAR}_2\approx 8.24\%,\qquad
\mathrm{EAR}_3\approx 8.21\%,\qquad
\mathrm{EAR}_4\approx 8.27\%.
\]

La mayor de las cuatro es
\[
\mathrm{EAR}_4.
\]
