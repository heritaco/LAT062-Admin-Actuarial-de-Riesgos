\subsection{EXAMPLE 6.3: FRM EXAM 2002---QUESTION 51 (time to double)}

\textbf{Enunciado}

\emph{A savings account pays an annual interest rate of 8\%. How long to double money?}
\begin{enumerate}[label=\alph*.]
\item 7 years
\item 8 years
\item \hl{9 years}
\item 10 years
\end{enumerate}

\subsubsection*{Solución}


\[
2=(1.08)^t \;\Rightarrow\; t=\frac{\ln 2}{\ln(1.08)}\approx 9.0.
\]
