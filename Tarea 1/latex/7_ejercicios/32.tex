\subsection{EXAMPLE 5.3: FRM EXAM 2002---QUESTION 3 (weekly volatility)}

\textbf{Enunciado}

\emph{Consider a stock with daily returns that follow a random walk. The annualized volatility is 34\%. Estimate the weekly volatility assuming that the year has 52 weeks.}
\begin{enumerate}[label=\alph*.]
\item 6.80\%
\item 5.83\%
\item 4.85\%
\item \hl{4.71\%}
\end{enumerate}

\subsubsection*{Solución}


\[
\sigma_{week}=\sigma_{ann}\sqrt{\frac{1}{52}}=\frac{0.34}{\sqrt{52}}
\approx \frac{0.34}{7.211}=0.0471=4.71\%.
\]
