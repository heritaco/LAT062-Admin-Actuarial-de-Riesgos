\begin{frame}{Propósito y hallazgos principales}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.47\textwidth}
    \smallhead{Alcance}
    \begin{itemize}
      \item Se comparan \textbf{Empresas, Entidades Financieras, Gubernamental y Vivienda}.
      \item Las dos fechas de referencia son \textbf{diciembre de 2022} y \textbf{diciembre de 2023}.
      \item Se revisan \textbf{IMOR, EL, UL, CaR} y \textbf{capital económico}.
    \end{itemize}

    \smallhead{Criterio conservador}
    Para cada cartera se usa el \textbf{CaR máximo} entre los escenarios con $\rho_1$ y $\rho_2$.
    El \textbf{capital económico} se presenta como \textbf{CaR máximo - EL}.
  \end{column}
  \begin{column}{0.49\textwidth}
    \smallhead{Cuatro mensajes}
    \begin{itemize}
      \item \textbf{Empresas} mejora en mora y pérdida esperada, pero su \textbf{CaR relativo} sube de 3.94\% a 4.82\%.
      \item \textbf{Entidades Financieras} conserva mora casi nula, aunque todavía exige \textbf{\$3,423 \mdp} de CaR.
      \item \textbf{Gubernamental} concentra el mayor consumo de capital económico: \textbf{\$23,183 \mdp} al cierre.
      \item \textbf{Vivienda} es la alerta central: IMOR en saldo de \textbf{2.52\% a 20.02\%} y CaR de \textbf{3.95\% a 14.75\%}.
    \end{itemize}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Marco de análisis}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.56\textwidth}
    \smallhead{Métricas consideradas}
    \begin{itemize}
      \item \textbf{EL}: pérdida esperada del portafolio.
      \item \textbf{UL}: desviación de pérdidas bajo dos correlaciones positivas.
      \item \textbf{CaR}: medida de pérdida extrema con $\theta = 5\%$ y $\alpha = 3$.
      \item \textbf{Capital económico}: capital requerido para absorber la pérdida no esperada extrema.
    \end{itemize}

    \smallhead{Interpretación}
    Cuando domina $\rho_1$, la fuente principal de riesgo es la \textbf{concentración dentro del mismo segmento}.
    Cuando domina $\rho_2$, el riesgo proviene sobre todo de un \textbf{choque sistémico transversal}.
  \end{column}
  \begin{column}{0.34\textwidth}
    \smallhead{Escenario vinculante}
    \scriptsize

      \midrule
      Empresas & 20\% & 5\% & $\rho_1$ \\
      Ent. Fin. & 40\% & 10\% & $\rho_1$ \\
      Gubernamental & 70\% & 30\% & $\rho_1$ \\
      Vivienda & 5\% & 60\% & $\rho_2$ \\
      \bottomrule
    \end{tabular}
  \end{column}
\end{columns}
\end{frame}

\vspace{0.25cm}
\smallhead{Conclusión comparativa}
La administración del riesgo no debe enfocarse en un solo síntoma: \textbf{Gubernamental} y
\textbf{Entidades Financieras} consumen capital por concentración; \textbf{Vivienda} exige acción
inmediata por deterioro ya materializado; \textbf{Empresas} mejoró, pero no debe relajarse el control.
\end{frame}

\begin{frame}{Recomendaciones inmediatas}
\small
\begin{enumerate}
  \item \textbf{Vivienda:} activar un plan de contención con cobranza intensiva, reestructuras focalizadas y seguimiento mensual a IMOR, LTV, cosecha y ubicación.
  \item \textbf{Gubernamental:} formalizar un buffer de capital alineado con un CaR cercano a \textbf{19\% del saldo} y realizar pruebas de estrés fiscales.
  \item \textbf{Entidades Financieras:} recalibrar límites por contraparte para que el crecimiento de saldo no se traduzca automáticamente en alto consumo de capital.
  \item \textbf{Empresas:} preservar la mejora de calidad y revisar concentración antes de acelerar nueva originación.
\end{enumerate}

\vspace{0.2cm}
\smallhead{Horizonte}
Estas acciones deben implementarse en los próximos \textbf{30 a 90 días}, priorizando primero
\textbf{Vivienda} y después \textbf{Gubernamental} y \textbf{Entidades Financieras}.
\end{frame}

\begin{frame}{Recomendaciones estructurales y cierre}
\small
\smallhead{Medidas estructurales}
\begin{itemize}
  \item Integrar un tablero trimestral con \textbf{IMOR, EL, UL, CaR y capital económico} por segmento.
  \item Definir un apetito de riesgo dual: \textbf{riesgo observado} y \textbf{riesgo de cola}.
  \item Vincular el crecimiento comercial a consumo de capital, no solo a morosidad histórica.
\end{itemize}

\smallhead{Conclusión final}
\begin{itemize}
  \item \textbf{Empresas} y \textbf{Entidades Financieras} mejoran en riesgo esperado.
  \item \textbf{Gubernamental} sigue siendo la principal fuente de consumo de capital económico.
  \item \textbf{Vivienda} requiere la reacción más rápida por el salto de IMOR y CaR.
\end{itemize}

\vspace{0.2cm}
\centering
\textbf{Decisión recomendada:} contener \textbf{Vivienda}, sostener disciplina en \textbf{Empresas}
y gestionar \textbf{Gubernamental} y \textbf{Entidades Financieras} con enfoque explícito de concentración y capital.
\end{frame}
