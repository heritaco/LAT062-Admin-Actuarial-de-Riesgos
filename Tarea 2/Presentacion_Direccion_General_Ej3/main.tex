\begin{frame}{Resumen ejecutivo}
\scriptsize
\fourkpigrid
  {Reservas requeridas}{USD 208,228.94}
  {CaR sin correlación}{USD 18,271,223.09}
  {CaR con correlación}{USD 32,706,941.92}
  {Impacto de correlación}{+79.0\%}

\vspace{0.36cm}
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.48\textwidth}
    \smallhead{Mensajes clave}
    \begin{itemize}\setlength{\itemsep}{0.25em}
      \item La cartera de \textbf{Asics, Toyota, Adidas y Volkswagen} exige \textbf{más capital} cuando se reconoce la correlación.
      \item \textbf{Toyota concentra 73.53\% de la pérdida esperada}; es el principal nombre a vigilar por exposición.
    \end{itemize}
  \end{column}
  \begin{column}{0.48\textwidth}
    \smallhead{Conclusión ejecutiva}
    \begin{itemize}\setlength{\itemsep}{0.25em}
      \item La \textbf{decisión de crédito no debe descansar en un solo modelo}; debe combinar solvencia contable, mercado y concentración.
      \item La mayor vulnerabilidad agregada proviene de la \textbf{concentración en Toyota} y del salto de capital bajo correlación.
    \end{itemize}

    \smallhead{Decisión sugerida}
    Aprobar con selectividad y usar \textbf{Altman + Merton + límites de concentración} como criterio conjunto.
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Alcance, universo y criterio}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.44\textwidth}
    \smallhead{Empresas analizadas}
    \begin{itemize}
      \item \textbf{Asia}: Asics y Toyota.
      \item \textbf{Europa}: Adidas y Volkswagen.
      \item La cartera se define como los \textbf{pasivos de largo plazo} de las cuatro compañías.
    \end{itemize}

    \smallhead{Conversión a USD}
    \begin{itemize}
      \item \textbf{EUR/USD = 1.1705}
      \item \textbf{USD/JPY = 159.52}
    \end{itemize}
  \end{column}
  \begin{column}{0.52\textwidth}
    \smallhead{Metodología aplicada}
    \begin{itemize}
      \item \textbf{Z-Altman}: estima calidad crediticia con base en razones financieras.
      \item \textbf{Merton}: estima PD implícita de mercado, valor de deuda, pérdida esperada, recuperación y LGD.
      \item \textbf{Cartera}: calcula pérdida esperada, pérdida no esperada, reservas y CaR con dos escenarios:
      \begin{itemize}
      \item correlación cero;
      \item correlación usando rendimientos accionarios como proxy.
      \end{itemize}
    \end{itemize}

    \smallhead{Parámetros de riesgo}
    \begin{itemize}
      \item \(\theta = 5\%\)
      \item \(\alpha = 2\)
    \end{itemize}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Conclusiones para Dirección General}
\small
\begin{enumerate}
  \item \textbf{Adidas} es el acreditado más limpio del ejercicio: buen Altman, PD mínima por Merton y LGD prácticamente nula.
  \item \textbf{Asics} es financiable, pero no libre de riesgo: su PD de Merton (\textbf{1.166\%}) es la más alta del grupo, aunque su score Altman sí aprueba.
  \item \textbf{Toyota} es el caso más delicado: Altman la rechaza, Merton la ve casi sin default y, además, \textbf{concentra 73.53\% de la pérdida esperada de la cartera}.
  \item \textbf{Volkswagen} presenta la misma tensión conceptual que Toyota: mala señal contable bajo Altman, pero señal benigna de mercado bajo Merton.
  \item A nivel cartera, el factor decisivo no es la mora aislada sino la \textbf{combinación de concentración + correlación}; por eso el CaR con correlación llega a \textbf{USD 32.71 millones}.
\end{enumerate}
\end{frame}

\begin{frame}{Recomendaciones inmediatas}
\footnotesize
\begin{enumerate}
  \item \textbf{No aprobar automáticamente} a Toyota ni Volkswagen con base en Merton; exigir comité, covenants y límites.
  \item \textbf{Mantener a Adidas} como referente de mejor calidad relativa dentro del grupo de análisis.
  \item \textbf{Autorizar Asics con seguimiento reforzado}, porque combina aprobación por Altman con la PD de mercado más alta.
  \item \textbf{Gestionar la cartera con capital económico correlacionado}, no con el escenario independiente.
\end{enumerate}

\vspace{0.18cm}
\smallhead{Sentido directivo}
\begin{itemize}
  \item La prioridad es evitar una lectura excesivamente optimista de Toyota y Volkswagen basada solo en mercado.
  \item La segunda prioridad es reconocer que la solvencia del portafolio depende del capital económico correlacionado.
\end{itemize}
\end{frame}

\begin{frame}{Medidas estructurales y cierre}
\footnotesize
\smallhead{Medidas estructurales}
\begin{itemize}
  \item Instituir una política dual de admisión: \textbf{filtro financiero (Altman)} más \textbf{validación de mercado (Merton)}.
  \item Establecer límites por nombre para impedir que Toyota explique más de 70\% de la pérdida esperada.
  \item Usar como referencia de solvencia un capital económico cercano a \textbf{USD 32.50 millones}, no solo la reserva de \textbf{USD 208 mil}.
  \item Recalcular trimestralmente correlaciones accionarias para actualizar el CaR.
\end{itemize}

\vspace{0.15cm}
\smallhead{Cierre}
\centering
\textbf{Decisión recomendada:} crecer solo con disciplina de selección y control explícito de concentración, porque el riesgo agregado aumenta fuertemente cuando las empresas se mueven juntas.
\end{frame}
