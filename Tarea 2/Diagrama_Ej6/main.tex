{\Large\bfseries Diagrama de procedimientos operativos: Portafolio de bonos con CreditMetrics}\\[1mm]
{\small Portafolio de tres bonos globales con calificaciones A-, BBB- y B-; vencimiento de 5 años; valor nominal USD \$100}
\end{center}

\vspace{2mm}

\begin{center}
\begin{adjustbox}{center,max totalsize={\textwidth}{0.82\textheight}}
\begin{tikzpicture}[node distance=4.5mm and 4mm]

% =========================================================
% BLOQUE SUPERIOR
% =========================================================

\node[startstop] (inicio) {Inicio};

\node[process, below=6mm of inicio] (carac) {
\textbf{Características de los bonos}\\
A-, BBB- y B-; vencimiento de 5 años; valor nominal USD \$100; cupón anual 7.5\%, 9\% y 11\%.
};

\node[section, below=6mm of carac] (inshead) {insumos CreditMetrics};

\node[process, below=5mm of inshead] (ins1) {
Matriz de transición a 1 año.
};

\node[process, below=4mm of ins1] (ins2) {
Curva de los bonos libre de riesgo.
};

\node[process, below=4mm of ins2, xshift=-24.5mm] (ins3t) {
Curva de los bonos por calidad crediticia.
};

\node[formula, right=4mm of ins3t] (ins3m) {
\[
\begin{aligned}
y_s &= y_f \\
    &\quad + \text{sobretasa}_s
\end{aligned}
\]
};

\node[process, below=5mm of ins3m, xshift=-49mm] (ins4t) {
Tasa de recuperación en caso de incumplimiento.
};

\node[formula, right=4mm of ins4t] (ins4m) {
\[
\begin{aligned}
RR_{A^-} &= 45\% \\
RR_{BBB^-} &= 35\% \\
RR_{B^-} &= 25\%
\end{aligned}
\]
};

\node[decision, below=7mm of ins4m, xshift=-24.5mm] (supuesto) {
Supuesto sobre probabilidades de incumplimiento
};

% =========================================================
% RAMA IZQUIERDA: INDEPENDENCIA
% =========================================================

\node[section, below left=13mm and 57mm of supuesto] (indhead) {
migraciones independientes
};

\node[process, below=5mm of indhead, xshift=-24.5mm] (ind1t) {
Valuar cada bono en cada estado de crédito.
};

\node[formula, right=7mm of ind1t] (ind1m) {
\[
\begin{aligned}
V_{i,s}
  &= \sum_{\tau=1}^{4}
     \frac{C_i}{(1+y_s)^\tau} \\
  &\quad + \frac{100+C_i}{(1+y_s)^4}
\end{aligned}
\]
\[
V_{i,D}=RR_i\times 100
\]
};

\node[process, below=5mm of ind1m, xshift=-52mm] (ind2t) {
Formar escenarios de migración y valuar el portafolio.
};

\node[formula, left=7mm of ind2t] (ind2m) {
\[
\omega=(s_1,s_2,s_3)
\]
\[
V_P(\omega)=V_{1,s_1}+V_{2,s_2}+V_{3,s_3}
\]
\[
m^3\ \text{escenarios}
\]
};

\node[process, below=5mm of ind2m, xshift=52mm] (ind3t) {
Si las migraciones fueran independientes.
};

\node[formula, right=7mm of ind3t] (ind3m) {
\[
\begin{aligned}
p(\omega)
  &= p_{A^-\to s_1}\,p_{BBB^-\to s_2} \\
  &\quad \times p_{B^-\to s_3}
\end{aligned}
\]
};

\node[process, below=7mm of ind3m, xshift=-52mm] (ind4t) {
Calcular media, desviación y pérdida por escenario.
};

\node[formula, left=7mm of ind4t] (ind4m) {
\[
\mu_P=\sum_{\omega}V_P(\omega)\,p(\omega)
\]
\[
\begin{aligned}
\sigma_P
  &= \Biggl[
     \sum_{\omega}
     \left(V_P(\omega)-\mu_P\right)^2 \\
  &\qquad \cdot p(\omega)
     \Biggr]^{1/2}
\end{aligned}
\]
\[
L(\omega)=V_{P,0}-V_P(\omega)
\]
};

\node[process, below=5mm of ind4m, xshift=52mm] (ind5t) {
\textbf{VaR}\\
Ordenar pérdidas y acumular probabilidades.
};

\node[formula, right=7mm of ind5t] (ind5m) {
\[
\begin{aligned}
VaR_{99.9\%}
  &= \inf\{\ell:\Pr(L\leq \ell) \\
  &\qquad \geq 0.999\}
\end{aligned}
\]
};

\node[output, below=6mm of ind5m, xshift=-26mm] (indout) {
\textbf{salida}\\
Distribución de pérdidas, pérdida esperada, desviación estándar y \(VaR_{99.9\%}\) bajo independencia.
};

% =========================================================
% RAMA DERECHA: DEPENDENCIA
% =========================================================

\node[section, below right=13mm and 57mm of supuesto] (dephead) {
migraciones dependientes
};

\node[process, below=5mm of dephead, xshift=-24.5mm] (dep1t) {
Modelar la correlación entre eventos de crédito.
};

\node[formula, left=7mm of dep1t] (dep1m) {
\[
R=
\begin{pmatrix}
1 & \rho_{12} & \rho_{13}\\
\rho_{21} & 1 & \rho_{23}\\
\rho_{31} & \rho_{32} & 1
\end{pmatrix}
\]
};

\node[process, below=5mm of dep1m, xshift=52mm] (dep2t) {
Transformar percentil a percentil.
};

\node[formula, right=7mm of dep2t] (dep2m) {
\[
x_i=N^{-1}\!\left(Q_i(t_i)\right)
\]
};

\node[process, below=5mm of dep2m, xshift=-52mm] (dep3t) {
Generar variables normales correlacionadas.
};

\node[formula, left=7mm of dep3t] (dep3m) {
\[
X=(x_1,x_2,x_3)^\top\sim N(0,R)
\]
};

\node[process, below=5mm of dep3m, xshift=52mm] (dep4t) {
Usar umbrales acumulados de la matriz de transición.
};

\node[formula, right=7mm of dep4t] (dep4m) {
\[
\sum_{k<s_i} p_{i,k}
<
N(x_i)
\leq
\sum_{k\leq s_i} p_{i,k}
\]
\[
\omega=(s_1,s_2,s_3)
\]
};

\node[process, below=7mm of dep4m, xshift=-52mm] (dep5t) {
Valuar el portafolio y construir pérdidas simuladas.
};

\node[formula, left=7mm of dep5t] (dep5m) {
\[
V_P(\omega)=V_{1,s_1}+V_{2,s_2}+V_{3,s_3}
\]
\[
L(\omega)=V_{P,0}-V_P(\omega)
\]
};

\node[process, below=5mm of dep5m, xshift=52mm] (dep6t) {
\textbf{VaR}\\
Ordenar pérdidas simuladas y tomar el percentil 99.9\%.
};

\node[formula, right=7mm of dep6t] (dep6m) {
\[
L_{(1)}\leq \cdots \leq L_{(M)}
\]
\[
VaR_{99.9\%}=L_{(\lceil 0.999M\rceil)}
\]
};

\node[output, below=6mm of dep6m, xshift=-26mm] (depout) {
\textbf{salida}\\
Distribución de pérdidas con correlación, pérdida esperada, desviación estándar y \(VaR_{99.9\%}\) bajo dependencia.
};

% =========================================================
% BLOQUE INFERIOR: n BONOS
% =========================================================

\coordinate (salidamid) at ($(indout.south)!0.5!(depout.south)$);

\node[section] (nhead) at ($(salidamid)+(0,-15mm)$) {
procedimiento para n bonos
};

\node[section, below left=8mm and 57mm of nhead] (ngenind) {
Con independencia
};

\node[process, below=5mm of ngenind, xshift=-24.5mm] (gen1t) {
Valuar cada bono, formar todos los escenarios y obtener la distribución de pérdidas del portafolio.
};

\node[formula, right=4mm of gen1t] (gen1m) {
\[
\omega=(s_1,\ldots,s_n)
\]
\[
\#\Omega=m^n
\]
\[
p(\omega)=\prod_{i=1}^{n}p_{i,s_i}
\]
};

\node[section, below right=8mm and 57mm of nhead] (ngendep) {
Con dependencia
};

\node[process, below=5mm of ngendep, xshift=-24.5mm] (gen2t) {
Estimar migraciones conjuntas, simular escenarios dependientes y valuar el portafolio.
};

\node[formula, right=4mm of gen2t] (gen2m) {
\[
X=(x_1,\ldots,x_n)^\top\sim N(0,R)
\]
\[
x_i=N^{-1}\!\left(Q_i(t_i)\right)
\]
\[
VaR_{99.9\%}
\]
};

\coordinate (genmid) at ($(gen1m.south)!0.5!(gen2m.south)$);

\node[note] (nota) at ($(genmid)+(0,-15mm)$) {
\textbf{Recordatorio}\\
CreditMetrics mide el riesgo de crédito por cambios de valor causados por migraciones en la calidad crediticia. Incorpora eventos de incumplimiento, mejoras y deterioros; además, permite evaluar diversificación y concentración de riesgo en la cartera.
};
