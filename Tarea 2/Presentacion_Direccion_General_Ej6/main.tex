\begin{frame}{Metodolog\'ia CreditMetrics}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.53\textwidth}
    \smallhead{L\'ogica del modelo}
    \begin{enumerate}
      \item Partir del \textbf{rating actual} de cada bono.
      \item Aplicar una \textbf{matriz de transici\'on a 1 a\~no}.
      \item Valuar el bono en cada \textbf{estado crediticio posible}, incluido default.
      \item Construir la \textbf{distribuci\'on de valor/p\'erdida} del portafolio.
      \item Reportar \textbf{E[V]}, \textbf{$\sigma$}, \textbf{VaR} y \textbf{CVaR}.
    \end{enumerate}

    \smallhead{M\'etrica de decisi\'on}
    \begin{itemize}
      \item \varlabel{VaR 99.9\%}: umbral de p\'erdida extrema.
      \item \cvarlabel{CVaR 99.9\%}: p\'erdida promedio en la cola que excede el VaR.
    \end{itemize}
  \end{column}
  \begin{column}{0.41\textwidth}
    \smallhead{Dos supuestos de dependencia}
    \footnotesize
    \begin{tabular}{L{1.75cm}L{2.85cm}}
      \toprule
      \rowcolor{udlaplight}

      \midrule
      Independiente & Producto de probabilidades marginales por bono. \\
      \rowcolor{softgray}

\begin{frame}{Interpretaci\'on del VaR 99.9\%}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.49\textwidth}
    \smallhead{Lectura correcta del VaR}
    \begin{itemize}
      \item \varlabel{VaR 1y 99.9\% = USD 211.2539} bajo independencia significa que
      s\'olo en el \textbf{0.1\%} de los escenarios la p\'erdida anual ser\'ia mayor a esa cifra.
      \item El \cvarlabel{CVaR = USD 230.1804} es la p\'erdida promedio en esa cola extrema.
      \item Por eso, el CVaR es el mejor complemento cuando la direcci\'on quiere entender
      \textbf{qu\'e tan costoso es fallar m\'as all\'a del VaR}.
    \end{itemize}
  \end{column}
  \begin{column}{0.45\textwidth}
    \smallhead{Implicaci\'on para decisiones}
    \begin{itemize}
      \item El promedio no protege contra cola; \textbf{el capital s\'i}.
      \item Un portafolio con ratings medios y bajos puede exhibir \textbf{p\'erdidas muy altas}
      aun con s\'olo tres nombres.
      \item Para Comit\'e o DG, el VaR 99.9\% funciona como \textbf{l\'imite de tolerancia}
      y el CVaR como \textbf{medida de severidad residual}.
    \end{itemize}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Procedimiento general para $n$ bonos}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.48\textwidth}
    \smallhead{Caso independiente}
    \begin{itemize}
      \item Definir para cada bono su conjunto de estados posibles $S_i$.
      \item Formar el \textbf{producto cartesiano}:
      \[
        \Omega = S_1 \times S_2 \times \cdots \times S_n
      \]
      \item Valuar cada escenario $\omega=(s_1,\dots,s_n)$.
      \item Obtener la probabilidad conjunta con el \textbf{producto de marginales}:
      \[
        p(\omega)=\prod_{i=1}^{n} p_i(s_i)
      \]
    \end{itemize}
  \end{column}
  \begin{column}{0.46\textwidth}
    \smallhead{Caso dependiente}
    \begin{itemize}
      \item Incorporar una \textbf{matriz, c\'opula o estructura de correlaci\'on}
      para migraciones conjuntas.
      \item Construir una \textbf{distribuci\'on conjunta calibrada} o simular
      escenarios dependientes.
      \item Valuar el portafolio en cada simulaci\'on y estimar la distribuci\'on de p\'erdidas.
      \item Reportar VaR/CVaR con base en la muestra conjunta:
      \[
        X \sim N(0,R)
      \]
    \end{itemize}
  \end{column}
\end{columns}

\end{frame}

\begin{frame}{Conclusiones}
\small
\begin{itemize}
  \item El ejercicio confirma que \textbf{CreditMetrics} traduce migraciones de rating en una
  distribuci\'on econ\'omica de valor y p\'erdida, no s\'olo en una probabilidad de default.
  \item Con las cifras reportadas, el supuesto \textbf{independiente} domina el consumo de capital:
  mayor VaR, mayor CVaR y mayor dispersi\'on.
  \item La versi\'on \textbf{correlacionada} mejora el valor esperado y reduce la cola,
  por lo que debe interpretarse como una \textbf{sensibilidad de calibraci\'on}, no como el caso prudencial base.
  \item La concentraci\'on en bonos con calidad A-/BBB-/B- y recuperaciones decrecientes genera una
  cola significativa aun en un portafolio peque\~no.
\end{itemize}
\end{frame}

\begin{frame}{Recomendaciones para Direcci\'on General}
\small
\begin{enumerate}
  \item Usar \textbf{VaR y CVaR independientes} como referencia principal para l\'imites y capital.
  \item Mantener el caso correlacionado como \textbf{an\'alisis de sensibilidad} y documentar claramente
  la calibraci\'on de dependencia antes de conceder alivios de capital.
  \item Revisar exposici\'on a nombres con \textbf{rating BBB- y B-}, porque son los que m\'as tensionan la cola.
  \item Complementar el monitoreo con \textbf{stress tests} de transici\'on y recuperaci\'on,
  especialmente en el percentil extremo de 99.9\%.
  \item Incorporar las m\'etricas escaladas de 1 y 10 d\'ias en el tablero gerencial para seguimiento continuo.
\end{enumerate}
\end{frame}
